Easyway to get selected option details

					
// add an onSelect callback
<select data-on-select="selected(event)"></select>

// get the selected value from list data-attributes
function selected(e){
	var $el = $(e.target);

	console.log('Selected Value: ' + $el.data('value'));
	console.log('Selected Index: ' + $el.data('index'));
	console.log('Selected Text: ' + $el.text());
}
